```typescript
export type TodoAssignee = 'tristan' | 'deltron';
export type TodoStatus = 'open' | 'done' | 'cancelled';
export type TodoPriority = 'low' | 'medium' | 'high';

export interface TodoProject {
  code: string;       // e.g. 'TSK', 'PFR'
  name: string;
  created_at: string;
}

export interface TodoItem {
  task_id: string;       // e.g. 'TSK-001', 'TSK-001-a'
  seq_num: number;
  title: string;
  assignee: TodoAssignee;
  status: TodoStatus;
  project_code: string;  // FK to todo_projects.code
  priority: TodoPriority | null;
  due_date: string | null;       // YYYY-MM-DD
  reminder_at: string | null;
  reminder_sent: number;
  tags: string | null;
  notes: string | null;
  notion_id: string | null;
  parent_task_id: string | null; // non-null = subtask
  subtask_letter: string | null; // 'a', 'b', etc.
  created_at: string;
  updated_at: string;
}

export function listTodos(filter?: { status?: TodoStatus; assignee?: TodoAssignee }): TodoItem[]
export function listProjects(): TodoProject[]
```

Note: `listTodos()` with no filter returns ALL items (open + done + cancelled).
Subtasks identified by: `parent_task_id !== null`.
Items ordered by: project_code, seq_num, subtask_letter (from listTodos SQL).

Router factory pattern:
```typescript
export function todosRouter(): Router {
  const router = Router();
  router.get('/todos', (_req, res) => {
    const items = listTodos();         // all statuses — client filters
    const projects = listProjects();
    res.json({ items, projects });
  });
  return router;
}
```
