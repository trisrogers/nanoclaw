    **In src/db.ts**, add after `logTaskRun`:
    ```typescript
    export function getTaskRunLogs(taskId: string, limit: number = 20): TaskRunLog[] {
      return getDb()
        .prepare(
          `SELECT task_id, run_at, duration_ms, status, result, error
           FROM task_run_logs WHERE task_id = ?
           ORDER BY run_at DESC LIMIT ?`
        )
        .all(taskId, Math.min(limit, 20)) as TaskRunLog[];
    }
    ```
    Hard cap at 20 (Math.min) — run logs accumulate indefinitely; TASK-02 requires max 20.

    **Create src/dashboard/routes/tasks.ts:**
    ```typescript
    import { Router } from 'express';
    import { getAllTasks, getTaskRunLogs } from '../../db.js';

    export function tasksRouter(): Router {
      const router = Router();

      router.get('/tasks', (_req, res) => {
        res.json(getAllTasks());
      });

      router.get('/tasks/:id/runs', (req, res) => {
        res.json(getTaskRunLogs(req.params.id));
      });

      return router;
    }
    ```
