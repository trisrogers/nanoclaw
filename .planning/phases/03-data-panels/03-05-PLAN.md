  <behavior>
    - GET /api/usage spawns `claude /usage`, returns { data: UsageData, error: null, fetchedAt: N }
    - Second GET within 60 seconds returns same cached response without re-spawning
    - When CLI fails (non-zero exit): returns { data: null, error: '<message>', fetchedAt: N } — never 500
    - When CLI not found (ENOENT): returns { data: null, error: 'claude CLI not found in PATH', fetchedAt: N }
    - When CLI succeeds but output cannot be parsed: returns { data: { raw: '<stdout>' }, error: null, fetchedAt: N }
    - fetchedAt > 60s ago: re-spawns CLI and updates cache
  </behavior>
  <action>

    ```bash

    ```
    Use the actual output to write `parseUsageOutput`. If the command isn't available in this environment, build a defensive parser that extracts numbers/text via regex and falls back to `{ raw: stdout }` when no pattern matches.

    **Create src/dashboard/routes/usage.ts:**

    ```typescript
    import { spawn } from 'child_process';
    import { Router } from 'express';

    interface UsageData {
      sessionUsage?: string;
      weeklyLimit?: string;
      resetTime?: string;
      raw?: string;
    }

    let usageCache: { data: UsageData | null; error: string | null; fetchedAt: number } | null = null;
    const CACHE_TTL_MS = 60_000;

    function parseUsageOutput(stdout: string): UsageData {
      // Defensive: try to extract known fields, fall back to raw
      // Adapt patterns based on empirical output from `claude /usage`
      const data: UsageData = { raw: stdout };
      // e.g. look for patterns like "Session: X tokens", "Weekly limit: Y", "Resets: Z"
      // Use regex, never throw — return partial data + raw
      return data;
    }

    function runClaudeUsage(): Promise<string> {
      return new Promise((resolve, reject) => {
        const chunks: Buffer[] = [];
        const proc = spawn('claude', ['/usage'], { timeout: 10_000 });
        proc.stdout.on('data', (chunk: Buffer) => chunks.push(chunk));
        proc.stderr.on('data', (chunk: Buffer) => chunks.push(chunk)); // capture stderr too
        proc.on('close', (code) => {
          if (code === 0) resolve(Buffer.concat(chunks).toString('utf8'));
          else reject(new Error(`claude /usage exited with code ${code}`));
        });
        proc.on('error', (err: NodeJS.ErrnoException) => {
          if (err.code === 'ENOENT') reject(new Error('claude CLI not found in PATH'));
          else reject(err);
        });
      });
    }

    export function usageRouter(): Router {
      const router = Router();
      router.get('/usage', async (_req, res) => {
        const now = Date.now();
        if (usageCache && now - usageCache.fetchedAt < CACHE_TTL_MS) {
          res.json(usageCache); return;
        }
        try {
          const stdout = await runClaudeUsage();
          const data = parseUsageOutput(stdout);
          usageCache = { data, error: null, fetchedAt: now };
        } catch (err) {
          usageCache = { data: null, error: String(err), fetchedAt: now };
        }
        res.json(usageCache);
      });
      return router;
    }
    ```
